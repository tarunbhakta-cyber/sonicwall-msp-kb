SECTION 1 – ESSENTIAL POWERSHELL PATTERNS

1.1 Remote Management

# Enable PSRemoting (run as admin on target)
Enable-PSRemoting -Force

# Remote session
Enter-PSSession -ComputerName SRV01 -Credential (Get-Credential)

# Run command on multiple machines
Invoke-Command -ComputerName SRV01,SRV02,SRV03 -ScriptBlock { Get-Service | Where Status -eq "Stopped" }

# Persistent session (reuse connection)
$session = New-PSSession -ComputerName SRV01
Invoke-Command -Session $session -ScriptBlock { Get-Process }
Remove-PSSession $session

# WinRM troubleshooting
winrm quickconfig
Test-WSMan -ComputerName SRV01
# If failing: Check WinRM service, firewall (5985/5986), TrustedHosts

1.2 Error Handling

try {

1.3 Output & Reporting

# Export to CSV
Get-ADUser -Filter * -Properties LastLogonDate |
    Select Name, SamAccountName, LastLogonDate, Enabled |
    Export-Csv -Path "C:\Reports\users.csv" -NoTypeInformation

SECTION 2 – ACTIVE DIRECTORY AUTOMATION

2.1 User Management

# Bulk user creation from CSV
Import-Csv "C:\newusers.csv" | ForEach-Object {
    New-ADUser -Name "$($_.FirstName) $($_.LastName)" `
        -GivenName $_.FirstName -Surname $_.LastName `
        -SamAccountName $_.Username -UserPrincipalName "$($_.Username)@domain.com" `
        -Path "OU=Users,DC=domain,DC=local" `
        -AccountPassword (ConvertTo-SecureString $_.Password -AsPlainText -Force) `
        -Enabled $true -ChangePasswordAtLogon $true
}

# Find inactive users (90+ days)
$cutoff = (Get-Date).AddDays(-90)
Get-ADUser -Filter {LastLogonDate -lt $cutoff -and Enabled -eq $true} -Properties LastLogonDate |
    Select Name, SamAccountName, LastLogonDate

# Find locked out users
Search-ADAccount -LockedOut | Select Name, SamAccountName, LockedOut

# Unlock all locked accounts
Search-ADAccount -LockedOut | Unlock-ADAccount

# Password expiration report
Get-ADUser -Filter {Enabled -eq $true -and PasswordNeverExpires -eq $false} -Properties "msDS-UserPasswordExpiryTimeComputed" |
    Select Name, @{N="ExpiryDate";E={[datetime]::FromFileTime($_."msDS-UserPasswordExpiryTimeComputed")}} |
    Where {$_.ExpiryDate -lt (Get-Date).AddDays(7)}

2.2 Computer Account Management

# Find stale computers (90+ days)
$cutoff = (Get-Date).AddDays(-90)
Get-ADComputer -Filter {LastLogonDate -lt $cutoff} -Properties LastLogonDate |
    Select Name, LastLogonDate, DistinguishedName

# Move computers to OU
Get-ADComputer -Filter {Name -like "WS-*"} | Move-ADObject -TargetPath "OU=Workstations,DC=domain,DC=local"

# Disable stale computers
Get-ADComputer -Filter {LastLogonDate -lt $cutoff -and Enabled -eq $true} -Properties LastLogonDate |
    Set-ADComputer -Enabled $false

2.3 Group Management

# Get group membership
Get-ADGroupMember -Identity "Domain Admins" -Recursive | Select Name, SamAccountName

# Find all groups a user belongs to
Get-ADPrincipalGroupMembership -Identity "jsmith" | Select Name

# Nested group membership audit
function Get-NestedGroupMember {
    param([string]$GroupName)
    Get-ADGroupMember $GroupName | ForEach-Object {
        if ($_.ObjectClass -eq "group") {
            Get-NestedGroupMember $_.Name
        } else { $_ }
    }
}
Get-NestedGroupMember "VPN-Users" | Select Name, SamAccountName -Unique

3.2 Common M365 Automation Scripts

# Bulk license assignment
$users = Get-Content "C:\users.txt"
$sku = Get-MgSubscribedSku | Where {$_.SkuPartNumber -eq "ENTERPRISEPACK"}
foreach ($user in $users) {
    Set-MgUserLicense -UserId $user -AddLicenses @{SkuId=$sku.SkuId} -RemoveLicenses @()
}

# Export all mailbox sizes
Get-Mailbox -ResultSize Unlimited | Get-MailboxStatistics |
    Select DisplayName, ItemCount, @{N="SizeMB";E={[math]::Round(($_.TotalItemSize.ToString().Split("(")[1].Split(" ")[0].Replace(",",""))/1MB,2)}} |
    Sort SizeMB -Descending | Export-Csv "C:\mailbox-sizes.csv" -NoTypeInformation

# Shared mailbox permissions audit
Get-Mailbox -RecipientTypeDetails SharedMailbox | ForEach-Object {
    $mb = $_.DisplayName
    Get-MailboxPermission $_.Identity | Where {$_.User -notlike "NT AUTHORITY*" -and $_.IsInherited -eq $false} |
        Select @{N="Mailbox";E={$mb}}, User, AccessRights
} | Export-Csv "C:\shared-mailbox-audit.csv" -NoTypeInformation

SECTION 4 – AZURE POWERSHELL / CLI

4.1 Azure Resource Management

# Start/stop VMs

New-AzSnapshot -ResourceGroupName "RG01" -SnapshotName "VM01-snap" -Snapshot $snapshot

5.2 Health Check Script Template

# Multi-server health check
$servers = "DC01","DC02","FS01","SQL01"
$results = foreach ($server in $servers) {
    $ping = Test-Connection $server -Count 1 -Quiet
    $services = Invoke-Command -ComputerName $server -ScriptBlock {
        Get-Service -Name DNS,NTDS,W32Time,DFSR | Select Name, Status
    } -ErrorAction SilentlyContinue
    [PSCustomObject]@{
        Server = $server
        Online = $ping
        Services = ($services | Where Status -ne "Running" | ForEach Name) -join ", "
    }
}
$results | Format-Table -AutoSize
